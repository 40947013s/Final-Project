void HPModify( Player player, int n,  xxx reason ) {
  if ( n < 0 ) {
    tmp = player.state;
    player.state = MINUS_HP;
    skill( &player, reason );
    player.state = tmp;
    player.hp += n;
  }
  else
    player.hp += n;

}
